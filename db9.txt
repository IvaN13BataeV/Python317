FROM kafedra
WHERE kafedra IN
    (SELECT [kafedra]
    FROM G
    WHERE [group] IN
        (SELECT [group]
        FROM student
        WHERE id_st=
            (select id_st
            from exzamen
            where predmet="Социология")))
